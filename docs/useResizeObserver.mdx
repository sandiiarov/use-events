# useResizeObserver

```js
import { useResizeObserver } from 'use-events';
```

```jsx
const Example = () => {
  const ref = React.useRef(null);
  const [width, height] = useResizeObserver(ref);

  return (
    <div ref={ref}>
      <b>width:</b> {width}
      <b>height:</b> {height}
    </div>
  );
};
```
