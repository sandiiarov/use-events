# useResizeObserver

```js
import { useResizeObserver } from 'use-events';
```

```jsx
const Example = () => {
  const ref = React.useRef(null);
  const { top, right, bottom, left, height, width, x, y } = useResizeObserver(
    ref
  );

  return (
    <ul ref={ref}>
      <li>top: {top}</li>
      <li>right: {right}</li>
      <li>bottom: {bottom}</li>
      <li>left: {left}</li>
      <li>height: {height}</li>
      <li>width: {width}</li>
      <li>x: {x}</li>
      <li>y: {y}</li>
    </ul>
  );
};
```
